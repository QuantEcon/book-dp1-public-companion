import numpy as np

α, λ, d, b = 0.01, 0.1, 0.02, 0.025
g = b - d
A = np.mat(np.array([[(1 - d) * (1 - λ) + b, (1 - d) * α + b], 
     [(1 - d) * λ, (1 - d) * (1 - α)]]))

ū = (1 + g - (1 - d) * (1 - α)) /  (1 + g - (1 - d) * (1 - α) + (1 - d) * λ)

ē = 1 - ū
x̄ = np.array([[ū], [ē]])

print(np.allclose(A * x̄, (1 + g) * x̄))   # prints true

def plot_paths(figname="../figures/lake_1.pdf", savefig=False):

    path_length = 100
